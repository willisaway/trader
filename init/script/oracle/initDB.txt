--创建用户
CREATE USER TRADER
  IDENTIFIED BY '1111'
  DEFAULT TABLESPACE TRADER
  TEMPORARY TABLESPACE TEMP
  PROFILE DEFAULT
  ACCOUNT UNLOCK;
  -- 1 Role for TRADER 
  GRANT DBA TO TRADER;
  ALTER USER TRADER DEFAULT ROLE ALL;
  -- 1 System Privilege for TRADER 

  GRANT UNLIMITED TABLESPACE TO TRADER;
